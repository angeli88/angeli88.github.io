class Solution {
    public int[] sortedSquares(int[] nums) {
        int len=nums.length;
        int[] res=new int[len];
        for(int i=0,j=len-1,pos=len-1;i<=j;pos--){
            if(Math.abs(nums[i])>Math.abs(nums[j])){
                res[pos]=nums[i]*nums[i];
                ++i;
            }else{
                res[pos]=nums[j]*nums[j];
                --j;
            }
        }
        return res;
    }
}
